      int ParseP(int P_y) {
        switch (lookahead) {
          case '(': {
            MatchToken('(');
            int P1_y = P_y;
            int P1_x = ParseP(P1_y);
            MatchToken(')');
            int P2_y = P1_x;
            int P2_x = ParseP(P2_y);
            return P1_x + P2_x;
          }
          case 'a': {
            return P_y + 1;
          }
          default:
            reportError();
        }
      }
      int ParseS() {
        switch (lookahead) {
          case '(':
          case 'a': {
            int P_y = 0;
            int P_x = ParseP(P_y);
            return P_x;
          }
          case '#': {
            return 0;
          }
          default: {
            reportError();
          }
        }
      }
